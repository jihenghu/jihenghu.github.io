无框中国地图

begin
cn=addfile("mlse.cn.cfr50.200308.nc","r")
mlses=cn->mlse

wks=gsn_open_wks("png","mlse.cn.cfr50.200308")
gsn_define_colormap(wks,"NCV_bright")

res=True
res@gsnFrame=False
res@gsnDraw=False

res@cnFillOn=True
res@cnLinesOn=False
res@cnFillMode="CellFill"
res@lbLabelBarOn=True
res@pmLabelBarOrthogonalPosF =-0.1
res@pmLabelBarWidthF =0.15
res@pmLabelBarHeightF =0.45
res@lbOrientation ="Vertical"

res@gsnAddCyclic=False
res@mpMinLatF = 18                  
res@mpMaxLatF = 54
res@mpMinLonF = 72
res@mpMaxLonF = 145

res@gsnLeftStringFontHeightF=0.03
res@mpGridAndLimbOn =False 
; res@tmXBBorderOn=False
; res@tmXTBorderOn=False
; res@tmYLBorderOn=False
; res@tmYRBorderOn=False
res@tmBorderLineColor="white"
res@tmXBOn=False
res@tmXTOn=False
res@tmYLOn=False
res@tmYROn=False

res@mpPerimOn=False
res@pmTickMarkDisplayMode  = "Never"

res@mpFillOn = True
res@mpMaskAreaSpecifiers = (/"China:states","Taiwan"/)
res@mpLandFillColor = "White"
res@mpOceanFillColor = "White"
res@mpInlandWaterFillColor = "White"
res@cnFillDrawOrder      = "PreDraw"

res@mpOutlineOn = True
res@mpOutlineBoundarySets = "NoBoundaries"           ; turn on country boundaries
res@mpGeophysicalLineColor      = "black"       ; color of cont. outlines
res@mpGeophysicalLineThicknessF = 1.0            ; thickness of outlines
res@mpNationalLineThicknessF = 2
res@mpDataBaseVersion = "MediumRes"              ; choose higher resolution
res@mpDataSetName = "Earth..4"                   ; choose most recent boundaries
res@mpOutlineSpecifiers = (/"China","Taiwan"/)        
res@lbLabelAutoStride = False
plot=new(3,graphic)
res@tmYLLabelsOn=True 
res@cnLevelSelectionMode="ExplicitLevels"
res@cnLevels:=(/0.88,0.90,0.92,0.93,0.94,0.945,0.95,0.955,0.96,0.97,0.98,0.99/);
res@lbLabelStrings:=(/"0.88","","0.92","","0.94","","0.95","","0.96","","0.98",""/);

res@gsnLeftString="(a) MLSE 10.65V"
plot(0)=gsn_csm_contour_map(wks,mlses(:,:,2),res)

res@tmYLLabelsOn=False 
res@gsnLeftString="(b) MLSE 18.7V"
plot(1)=gsn_csm_contour_map(wks,mlses(:,:,4),res)

res@gsnLeftString="(c) MLSE 36.5V"
plot(2)=gsn_csm_contour_map(wks,mlses(:,:,8),res)

;--- add South China Sea --- 
nhres                          = res
nhres@gsnMaximize              = False

nhres@vpHeightF                = 0.12    
nhres@vpWidthF                 = 0.12

nhres@mpMinLatF                :=   2.0    
nhres@mpMaxLatF                :=  23.0
nhres@mpMinLonF                := 105.0
nhres@mpMaxLonF                := 123.0
nhres@tmBorderThicknessF=1.
nhres@tmBorderLineColor="black"
nhres@tmXBBorderOn=True
nhres@tmXTBorderOn=True
nhres@tmYLBorderOn=True
nhres@tmYRBorderOn=True
nhres@lbLabelBarOn             = False
nhres@tmXBOn                   = False 
nhres@tmXTOn                   = False
nhres@tmYLOn                   = False
nhres@tmYROn                   = False
nhres@gsnLeftString            = ""
nhres@gsnRightString           = ""
nhres@tiMainString       = ""
;;
map_nanhai0 = gsn_csm_contour_map(wks,mlses(:,:,2),nhres)
map_nanhai1 = gsn_csm_contour_map(wks,mlses(:,:,4),nhres)
map_nanhai2 = gsn_csm_contour_map(wks,mlses(:,:,8),nhres)

adres                          = True
adres@amParallelPosF           = 0.37 ; -0.5 is the left edge of the plot.
adres@amOrthogonalPosF         = 0.49  ; -0.5 is the top edge of the plot.
adres@amJust                   = "BottomRight"
;;
plotnh0 = gsn_add_annotation(plot(0),map_nanhai0,adres)
plotnh1 = gsn_add_annotation(plot(1),map_nanhai1,adres)
plotnh2 = gsn_add_annotation(plot(2),map_nanhai2,adres)

resP=True
gsn_panel(wks,plot,(/1,3/),resP)
end
