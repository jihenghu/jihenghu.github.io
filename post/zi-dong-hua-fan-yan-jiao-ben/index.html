自动化反演脚本

脚本的内容：

#!/bin/bash
first=$1
second=$2
while [ "$first" -le "$second" ]
do
    echo + [thread-main] $first  start ++++++++++++++++++++++++++++++++++
	    ## 2a -------------------------------------
		echo + [thread-2A] ---------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/2a
		ncl -Q yyyymmdd=${first} preprocess.2a.ncl
		
		## rain ------------------------------------
		echo + [thread-Rain] ---------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/rain
		ncl -Q yyyymmdd=${first} preprocess.rain.ncl
				
		## Modis ------------------------------------
		echo + [thread-Cloud] ----------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/06
		ncl -Q yyyymmdd=${first} myd06.preprocess.ncl
		
	    ## ERA 20C -------------------------------------
		echo + [thread-ERA] --------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/20c
		# rm -v era20c.easia.37L.125deg.${first}
		ncl -Q yyyymmdd=${first} era20c.preprocess.ncl
		
		## Collo ------------------------------------
		echo + [thread-Collo] --------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/collocate
		./collocate.exe $first
		
		## remove era preprocess data. too large (~700MB)		
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/20c/data
		rm -v era20c.easia.37L.125deg.${first}
		## remove myd06 data. too large (~100MB)		
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/06/data
		rm -v MYD06.cloud.properties.res5km.${first}.txt
		## retrieve ------------------------------------
		echo + [thread-Retr] --------------------------
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/retrieve
		./retrieve.exe $first
		
		## remove collocate profiles too large (300MB)
		cd /data01/hjh/mlse.AMSRE.CN.V2020/res1/collocate/data
		rm -v ael2a.ael2b.myd06.era20c.profile.${first}.txt
		rm -v ael2a.ael2b.myd06.era20c.hydro.${first}.txt
		rm -v ael2a.ael2b.myd06.era20c.basic.${first}.txt
		
    echo + [thread-main] $first finished
    let first=`date -d "-1 days ago ${first}" +%Y%m%d`
done
echo + [thread-main] all days done
